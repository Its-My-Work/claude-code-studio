**GET /api/sessions/:id/export**
- Look up session with `stmts.getSession.get(id)`. Return 404 if not found.
- Fetch all messages with `stmts.getMsgs.all(id)` (full content, not lite).
- Return JSON: `{ version: 1, exported_at: new Date().toISOString(), session: { ...sessionRow }, messages: [...messageRows] }`.
- Set header `Content-Disposition: attachment; filename="session-<id>.json"` and `Content-Type: application/json`.

**POST /api/sessions/import**
- Accept JSON body: `{ session, messages }` — same shape as the export format.
- Validate: `session` must be an object, `messages` must be an array. Return 400 otherwise.
- Generate a fresh ID with `genId()` for the new session.
- Insert the new session:
  ```js
  stmts.createSession.run(
    newId,
    String(session.title || 'Imported session').substring(0, 200),
    session.active_mcp || '[]',
    session.active_skills || '[]',
    session.mode || 'auto',
    session.agent_mode || 'single',
    session.model || 'sonnet',
    session.workdir || null
  );
  ```
- Set `claude_session_id = null` explicitly: `db.prepare("UPDATE sessions SET claude_session_id=NULL WHERE id=?").run(newId);`
- Insert all messages in a transaction. For each message (limit to 2000 messages max to prevent abuse):
  ```js
  stmts.addMsg.run(newId, m.role, m.type, m.content || '', m.tool_name || null, m.agent_id || null, null, null);
  ```
- Return `{ ok: true, session: stmts.getSession.get(newId) }` with status 201.

Both endpoints are synchronous (no async needed). Wrap inserts in `db.transaction()` for atomicity on import.

```javascript
async function exportSession() {
  if (!currentSessionId) return;
  try {
    const r = await fetch(`/api/sessions/${currentSessionId}/export`);
    if (!r.ok) { showToast('Export failed: ' + r.status, 'error'); return; }
    const blob = await r.blob();
    const url = URL.createObjectURL(blob);
    const a = document.createElement('a');
    a.href = url;
    a.download = `session-${currentSessionId}.json`;
    a.click();
    URL.revokeObjectURL(url);
  } catch (e) { showToast('Export error: ' + e.message, 'error'); }
}

async function importSessionFile(input) {
  const file = input.files[0];
  if (!file) return;
  input.value = '';
  try {
    const text = await file.text();
    const data = JSON.parse(text);
    if (!data.session || !Array.isArray(data.messages)) {
      showToast('Invalid export file', 'error'); return;
    }
    const r = await fetch('/api/sessions/import', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ session: data.session, messages: data.messages })
    });
    if (!r.ok) { showToast('Import failed: ' + r.status, 'error'); return; }
    const { session } = await r.json();
    showToast('Session imported', 'success');
    await loadHist();
    openTab(session.id, session.title);
  } catch (e) { showToast('Import error: ' + e.message, 'error'); }
}
```

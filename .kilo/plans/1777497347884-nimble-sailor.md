# План реализации отображения списка MCP kilo и их статуса

## Цель
Заменить существующий раздел MCP в веб-интерфейсе Claude Code Studio на отображение списка MCP-серверов kilo с их статусом подключения, аналогично выводу команды `kilo mcp list`.

## Текущая ситуация
- Веб-интерфейс показывает MCP-серверы Claude из `config.json`
- Команда `kilo mcp list` показывает MCP-серверы kilo с статусом подключения
- Нет интеграции между kilo MCP и веб-интерфейсом

## Решения на основе уточнений пользователя
- **Интеграция:** Заменить существующий раздел MCP на kilo MCP
- **Обновление статуса:** Обновлять при каждом открытии раздела
- **Обработка ошибок:** Показывать пустой раздел с сообщением если kilo недоступен

```javascript
app.get('/api/kilo-mcp-list', async (req, res) => {
  try {
    const { execSync } = require('child_process');
    const output = execSync('kilo mcp list', { encoding: 'utf-8' });
    const servers = parseKiloMcpOutput(output);
    res.json({ servers, error: null });
  } catch (error) {
    console.error('Kilo MCP list error:', error.message);
    res.json({ servers: [], error: 'Kilo is not available or MCP command failed' });
  }
});
```

#### 1.2 Создать функцию парсера вывода
```javascript
function parseKiloMcpOutput(output) {
  const servers = [];
  const lines = output.split('\n');

  // Найти начало списка серверов (после заголовка)
  let inServerList = false;
  let currentServer = null;

  for (const line of lines) {
    // Пропустить заголовок и разделители
    if (line.includes('MCP Servers') || line.includes('┌') || line.includes('└')) {
      continue;
    }

    // Начало сервера: строка с ● или │
    if (line.includes('●') || line.includes('│')) {
      if (line.includes('●')) {
        // Новая строка сервера
        const statusMatch = line.match(/●\s+([✓✗])\s+(\w+)\s+(\w+)/);
        if (statusMatch) {
          const [, statusIcon, name, status] = statusMatch;
          currentServer = {
            name,
            status: statusIcon === '✓' ? 'connected' : 'disconnected',
            command: '',
            enabled: statusIcon === '✓'
          };
          servers.push(currentServer);
        }
      } else if (currentServer && line.includes('│') && line.includes('node')) {
        // Команда запуска
        const cmdMatch = line.match(/│\s+(.+)/);
        if (cmdMatch) {
          currentServer.command = cmdMatch[1].trim();
        }
      }
    }
  }

  return servers;
}
```

#### 2.2 Модифицировать renderMcp()
```javascript
async function renderMcp() {
  const el = $i('mcpList');
  el.innerHTML = '';

  // Получить список kilo MCP
  const { servers, error } = await fetchKiloMcpList();

  if (error) {
    // Показать сообщение об ошибке
    el.innerHTML = `<div class="error-msg" style="padding:10px;color:#ff6b6b;text-align:center;">
      ${error}<br>
      <small style="color:#999;">Kilo MCP servers unavailable</small>
    </div>`;
    $i('mcpCount').textContent = '0';
    return;
  }

  let activeCount = 0;

  for (const server of servers) {
    const isActive = server.status === 'connected';
    if (isActive) activeCount++;

    const div = document.createElement('div');
    div.className = `ci ${isActive ? 'on' : ''}`;
    div.title = isActive ? 'Connected' : 'Disconnected';
    div.setAttribute('role', 'switch');
    div.setAttribute('tabindex', '0');
    div.setAttribute('aria-checked', String(isActive));
    div.setAttribute('aria-label', server.name);

    div.innerHTML = `
      <div class="toggle">${isActive ? '●' : '○'}</div>
      <div class="inf">
        <div class="nm">${escH(server.name)}</div>
        <div class="ds">${escH(server.command || 'No command')}</div>
        <div class="st">${isActive ? 'Connected' : 'Disconnected'}</div>
      </div>
      ${isActive ? '<div class="status-dot" style="color:#4CAF50;">✓</div>' : '<div class="status-dot" style="color:#ff6b6b;">✗</div>'}
    `;

    el.appendChild(div);
  }

  $i('mcpCount').textContent = activeCount;
  _reapplySecFilter('mcp');
}
```
